my_cumsum = function(arg) {
  result = c(arg[1])
  for(i in 2:length(arg))
    result <- c(result, tail(result, 1) + arg[i])
  return(result)
}

my_cumprod = function(arg) {
  result = c(arg[1])
  for(i in 2:length(arg))
    result <- c(result, tail(result, 1) * arg[i])
  return(result)
}
